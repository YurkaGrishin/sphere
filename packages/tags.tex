\newcommand{\ProcessTags}{
  \ifx\TaskSetClassTag\undefined
    \usetag{
      PrintProblemAuthors,%
      PrintSolutions,%
      PrintSmartPrefix,%
      PrintAllProblems,%
    }
  \else
    \usetag{\TaskSetClassTag}
    \TaskSetInfo{
        group=\TaskSetClassTag\space класс
    }
    \tagged{TypeSolutions}{
        \usetag{%
            PrintProblemAuthors,%
            PrintSolutions,%
            PrintSmartPrefix,%
        }
        \droptag{%
            PrintAllProblems,%
        }
        \TaskSetInfo{
            descr=Условия{,} решения и схема оценивания. \TaskSetGroup%
        }
    }

    \tagged{TypeProblems}{
        \droptag{%
            PrintProblemAuthors,%
            PrintAllProblems,%
            PrintSolutions,%
            PrintSmartPrefix,%
        }
        \TaskSetInfo{
            descr=Условия задач. \TaskSetGroup%
        }
    }
  \fi
}
